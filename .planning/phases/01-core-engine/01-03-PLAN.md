# Image Format Handling from PITFALLS.md

Common pitfalls to avoid:
1. "Assuming all responses are JPEG" - Bing/Spotlight APIs may return WebP or HEIC
2. "Not handling large images" - Set image::io::Limits to prevent OOM
3. "Image decode errors" - Handle image::ImageError explicitly

Mitigations required:
- Set image::io::Limits (max width 8192, max image size 50MB)
- Support JPEG, PNG, WebP, HEIC formats
- Validate image header before full decode
- Handle corrupted images gracefully

# LRU Eviction Strategy

From CORE-05 requirements:
- Max 500MB total size
- Max 50 images
- Max 30 days old

Eviction order:
1. Remove images older than 30 days first
2. If still over size/count limits, remove least recently accessed (LRU)
3. Track access time via file atime or metadata

# Image Format Support

Supported formats (from research):
- JPEG - Most common, well-supported by image crate
- PNG - Standard format, well-supported
- WebP - Modern format, may be returned by Bing/Spotlight
- HEIC - Apple format, may be returned by Spotlight (requires libheif)

Note: HEIC support may require additional dependencies (libheif-sys). For MVP, we'll handle HEIC as "unsupported format" with clear error message.

use std::path::{Path, PathBuf};
use std::time::{Duration, SystemTime, UNIX_EPOCH};
use image::{ImageFormat, DynamicImage};
use crate::wallpaper::Wallpaper;
use crate::error::CacheError;

/// Cache configuration
#[derive(Debug, Clone)]
pub struct CacheConfig {
    /// Maximum cache size in bytes (default: 500MB)
    pub max_size_bytes: u64,
    /// Maximum number of cached images (default: 50)
    pub max_count: usize,
    /// Maximum age of cached images in seconds (default: 30 days)
    pub max_age_seconds: u64,
}

impl Default for CacheConfig {
    fn default() -> Self {
        Self {
            max_size_bytes: 500 * 1024 * 1024, // 500MB
            max_count: 50,
            max_age_seconds: 30 * 24 * 60 * 60, // 30 days
        }
    }
}

/// Cache entry metadata
#[derive(Debug, Clone)]
struct CacheEntry {
    id: String,
    path: PathBuf,
    size_bytes: u64,
    accessed_at: u64, // Unix timestamp
    created_at: u64, // Unix timestamp
}

/// Image cache manager with LRU eviction
pub struct Cache {
    cache_dir: PathBuf,
    config: CacheConfig,
    entries: Vec<CacheEntry>,
}

impl Cache {
    /// Create a new cache manager
    ///
    /// Creates cache directory at `~/.cache/damask/` if it doesn't exist
    pub fn new(config: CacheConfig) -> Result<Self, CacheError> {
        let cache_dir = dirs::cache_dir()
            .ok_or(CacheError::NotAccessible)?
            .join("damask");

        // Create cache directory if it doesn't exist
        std::fs::create_dir_all(&cache_dir)
            .map_err(|e| CacheError::IoError(e))?;

        Ok(Self {
            cache_dir,
            config,
            entries: Vec::new(),
        })
    }

    /// Create a new cache manager with default configuration
    pub fn with_defaults() -> Result<Self, CacheError> {
        Self::new(CacheConfig::default())
    }

    /// Get the cache directory path
    pub fn cache_dir(&self) -> &Path {
        &self.cache_dir
    }

    /// Get cache configuration
    pub fn config(&self) -> &CacheConfig {
        &self.config
    }

    /// Load existing cache entries from disk
    pub async fn load_entries(&mut self) -> Result<(), CacheError> {
        self.entries.clear();

        let mut read_dir = tokio::fs::read_dir(&self.cache_dir).await
            .map_err(|e| CacheError::IoError(e))?;

        while let Some(entry) = read_dir.next_entry().await
            .map_err(|e| CacheError::IoError(e))?
        {
            let path = entry.path();

            // Only process files (not directories)
            if path.is_file() {
                // Get file metadata
                let metadata = tokio::fs::metadata(&path).await
                    .map_err(|e| CacheError::IoError(e))?;

                let modified = metadata.modified()
                    .map_err(|e| CacheError::IoError(e))?;

                let created = metadata.created()
                    .unwrap_or(modified); // Use modified time if created time unavailable

                let accessed = modified; // Use modified time as proxy for access time

                let size_bytes = metadata.len();

                // Extract ID from filename (remove extension)
                let id = path.file_stem()
                    .and_then(|s| s.to_str())
                    .ok_or(CacheError::NotAccessible)?
                    .to_string();

                self.entries.push(CacheEntry {
                    id,
                    path,
                    size_bytes,
                    accessed_at: accessed.duration_since(UNIX_EPOCH)
                        .map_err(|e| CacheError::IoError(e))?
                        .as_secs(),
                    created_at: created.duration_since(UNIX_EPOCH)
                        .map_err(|e| CacheError::IoError(e))?
                        .as_secs(),
                });
            }
        }

        Ok(())
    }

    /// Get or download a wallpaper
    ///
    /// Returns local file path if cached, downloads and caches otherwise
    pub async fn get_or_download(&mut self, wallpaper: &Wallpaper) -> Result<PathBuf, CacheError> {
        // Load entries if not already loaded
        if self.entries.is_empty() {
            self.load_entries().await?;
        }

        // Check if already cached
        if let Some(entry) = self.find_entry(&wallpaper.id) {
            // Update access time (LRU)
            self.update_access_time(&wallpaper.id);
            return Ok(entry.path.clone());
        }

        // Download and cache
        self.download_and_cache(wallpaper).await
    }

    /// Find a cache entry by ID
    fn find_entry(&self, id: &str) -> Option<&CacheEntry> {
        self.entries.iter().find(|e| e.id == id)
    }

    /// Update access time for an entry (LRU tracking)
    fn update_access_time(&mut self, id: &str) {
        if let Some(entry) = self.entries.iter_mut().find(|e| e.id == id) {
            entry.accessed_at = SystemTime::now()
                .duration_since(UNIX_EPOCH)
                .unwrap_or_default()
                .as_secs();
        }
    }

    /// Download and cache a wallpaper
    async fn download_and_cache(&mut self, wallpaper: &Wallpaper) -> Result<PathBuf, CacheError> {
        // Download image bytes
        let client = reqwest::Client::new();
        let response = client.get(&wallpaper.url)
            .send()
            .await
            .map_err(|e| CacheError::IoError(std::io::Error::new(
                std::io::ErrorKind::Other,
                format!("HTTP request failed: {}", e)
            )))?;

        if !response.status().is_success() {
            return Err(CacheError::IoError(std::io::Error::new(
                std::io::ErrorKind::Other,
                format!("HTTP error: {}", response.status())
            )));
        }

        let image_bytes = response.bytes().await
            .map_err(|e| CacheError::IoError(std::io::Error::new(
                std::io::ErrorKind::Other,
                format!("Failed to download image: {}", e)
            )))?;

        // Validate image format and decode to ensure it's a valid image
        let _validated_image = self.validate_image(&image_bytes)?;

        // Determine file extension from image format
        let extension = self.detect_extension(&image_bytes)?;

        // Generate cache file path
        let cache_path = self.cache_dir.join(format!("{}.{}", wallpaper.id, extension));

        // Write image bytes to cache
        tokio::fs::write(&cache_path, &image_bytes).await
            .map_err(|e| CacheError::IoError(e))?;

        // Get file metadata for size and timestamps
        let metadata = tokio::fs::metadata(&cache_path).await
            .map_err(|e| CacheError::IoError(e))?;

        let created = metadata.created()
            .unwrap_or(SystemTime::now());

        let accessed = metadata.modified()
            .unwrap_or(SystemTime::now());

        // Create cache entry
        self.entries.push(CacheEntry {
            id: wallpaper.id.clone(),
            path: cache_path.clone(),
            size_bytes: metadata.len(),
            accessed_at: accessed.duration_since(UNIX_EPOCH)
                .map_err(|e| CacheError::IoError(e))?
                .as_secs(),
            created_at: created.duration_since(UNIX_EPOCH)
                .map_err(|e| CacheError::IoError(e))?
                .as_secs(),
        });

        // Evict if over limits
        self.evict_if_needed().await?;

        Ok(cache_path)
    }

    /// Validate image format and decode to ensure it's a valid image
    fn validate_image(&self, image_bytes: &[u8]) -> Result<DynamicImage, CacheError> {
        // Set limits to prevent OOM on large or corrupted images
        let limits = image::io::Limits::default()
            .max_image_width(Some(8192))
            .max_image_height(Some(8192))
            .max_alloc(Some(50 * 1024 * 1024)); // 50MB

        // Try to decode image with format detection
        let image = image::load_from_memory_with_format(image_bytes, ImageFormat::Jpeg)
            .or_else(|_| image::load_from_memory_with_format(image_bytes, ImageFormat::Png))
            .or_else(|_| image::load_from_memory_with_format(image_bytes, ImageFormat::WebP))
            .map_err(|e| CacheError::ImageError(e))?;

        Ok(image)
    }

    /// Detect image format and return appropriate file extension
    fn detect_extension(&self, image_bytes: &[u8]) -> Result<&'static str, CacheError> {
        // Try to detect format by attempting to decode
        if image::load_from_memory_with_format(image_bytes, ImageFormat::Jpeg).is_ok() {
            return Ok("jpg");
        }

        if image::load_from_memory_with_format(image_bytes, ImageFormat::Png).is_ok() {
            return Ok("png");
        }

        if image::load_from_memory_with_format(image_bytes, ImageFormat::WebP).is_ok() {
            return Ok("webp");
        }

        // HEIC support requires libheif - for now, reject as unsupported
        // TODO: Add HEIC support in future version if needed
        return Err(CacheError::ImageError(image::ImageError::IoError(
            std::io::Error::new(
                std::io::ErrorKind::InvalidData,
                "Unsupported image format (HEIC not supported in MVP)"
            )
        )));
    }

    /// Evict cache entries if over limits
    async fn evict_if_needed(&mut self) -> Result<(), CacheError> {
        // Calculate current cache size and count
        let total_size: u64 = self.entries.iter().map(|e| e.size_bytes).sum();
        let count = self.entries.len();

        // Check if eviction needed
        let mut needs_eviction = false;

        if total_size > self.config.max_size_bytes {
            eprintln!("Cache size {}MB exceeds limit {}MB, evicting...",
                total_size / (1024 * 1024),
                self.config.max_size_bytes / (1024 * 1024)
            );
            needs_eviction = true;
        }

        if count > self.config.max_count {
            eprintln!("Cache count {} exceeds limit {}, evicting...",
                count,
                self.config.max_count
            );
            needs_eviction = true;
        }

        if !needs_eviction {
            return Ok(());
        }

        // Sort entries by (created_at, accessed_at) for eviction
        // First, remove entries older than max_age_seconds
        let now = SystemTime::now()
            .duration_since(UNIX_EPOCH)
            .unwrap_or_default()
            .as_secs();

        let old_entries: Vec<_> = self.entries
            .iter()
            .filter(|e| now - e.created_at > self.config.max_age_seconds)
            .map(|e| e.id.clone())
            .collect();

        for entry_id in old_entries {
            self.remove_entry(&entry_id).await?;
        }

        // Recalculate after removing old entries
        let total_size: u64 = self.entries.iter().map(|e| e.size_bytes).sum();
        let count = self.entries.len();

        // If still over limits, remove least recently accessed (LRU)
        if total_size > self.config.max_size_bytes || count > self.config.max_count {
            self.evict_lru_until_within_limits().await?;
        }

        Ok(())
    }

    /// Remove least recently accessed entries until within limits
    async fn evict_lru_until_within_limits(&mut self) -> Result<(), CacheError> {
        loop {
            // Calculate current cache size and count
            let total_size: u64 = self.entries.iter().map(|e| e.size_bytes).sum();
            let count = self.entries.len();

            // Check if within limits
            if total_size <= self.config.max_size_bytes && count <= self.config.max_count {
                break;
            }

            // Find and remove least recently accessed entry
            if let Some(lru_entry) = self.entries
                .iter()
                .min_by_key(|e| e.accessed_at)
            {
                self.remove_entry(&lru_entry.id).await?;
            } else {
                // No entries left to remove
                break;
            }
        }

        Ok(())
    }

    /// Remove a cache entry
    async fn remove_entry(&mut self, id: &str) -> Result<(), CacheError> {
        // Find entry
        if let Some(pos) = self.entries.iter().position(|e| e.id == id) {
            let entry = self.entries.remove(pos);

            // Delete file
            tokio::fs::remove_file(&entry.path).await
                .map_err(|e| CacheError::IoError(e))?;

            eprintln!("Evicted cache entry: {} ({}MB)",
                entry.id,
                entry.size_bytes / (1024 * 1024)
            );
        }

        Ok(())
    }

    /// Manually trigger cache cleanup
    pub async fn cleanup(&mut self) -> Result<usize, CacheError> {
        let initial_count = self.entries.len();

        self.evict_if_needed().await?;

        let removed_count = initial_count - self.entries.len();

        Ok(removed_count)
    }

    /// Get cache statistics
    pub fn stats(&self) -> CacheStats {
        let total_size: u64 = self.entries.iter().map(|e| e.size_bytes).sum();
        let count = self.entries.len();

        CacheStats {
            total_size_bytes: total_size,
            total_size_mb: total_size / (1024 * 1024),
            count,
            max_size_bytes: self.config.max_size_bytes,
            max_count: self.config.max_count,
        }
    }
}

/// Cache statistics
#[derive(Debug, Clone)]
pub struct CacheStats {
    pub total_size_bytes: u64,
    pub total_size_mb: u64,
    pub count: usize,
    pub max_size_bytes: u64,
    pub max_count: usize,
}

impl Default for Cache {
    fn default() -> Result<Self, CacheError> {
        Self::with_defaults()
    }
}
