    **Step E: Create app/src/widgets/settings.rs -- the SettingsComponent:**

    This is a standalone `Component` (not AsyncComponent -- no async needed for a settings dialog). It uses `relm4::Component` trait with a simple model, view, and update.

    ```rust
    use relm4::{Component, ComponentParts, ComponentSender, SimpleComponent};
    use gtk::prelude::*;
    use adw::prelude::*;

    pub struct SettingsModel {
        bing_market: String,
        spotlight_locale: String,
    }

    #[derive(Debug)]
    pub enum SettingsInput {
        Show,
    }

    #[derive(Debug)]
    pub enum SettingsOutput {
        SettingsChanged { bing_market: String, spotlight_locale: String },
    }

    pub struct SettingsComponent;

    impl SimpleComponent for SettingsComponent {
        type Init = (String, String);  // (bing_market, spotlight_locale)
        type Input = SettingsInput;
        type Output = SettingsOutput;
        type Root = adw::PreferencesWindow;

        fn init_root() -> Self::Root {
            adw::PreferencesWindow::new()
        }

        fn init(
            (bing_market, spotlight_locale): Self::Init,
            root: Self::Root,
            sender: ComponentSender<Self>,
        ) -> ComponentParts<Self> {
            let model = SettingsModel {
                bing_market,
                spotlight_locale,
            };

            // Build Bing market options
            let markets = [
                "en-US", "zh-CN", "ja-JP", "en-GB", "fr-FR", "de-DE",
                "pt-BR", "es-ES", "it-IT", "ru-RU",
            ];
            let market_names = [
                "English (US)", "Chinese (Simplified)", "Japanese", "English (UK)",
                "French", "German", "Portuguese (Brazil)", "Spanish", "Italian", "Russian",
            ];
            let market_list = gtk::StringList::new(&market_names);
            // Set initial selection based on current bing_market value
            let initial_index = markets.iter().position(|&m| m == model.bing_market).unwrap_or(0) as u32;

            // Build Spotlight locale entry
            let locale_entry = adw::EntryRow::new();
            locale_entry.set_title("Locale");
            locale_entry.set_text(&model.spotlight_locale);

            // Build widget tree using manual gtk calls (not view! macro -- PreferencesWindow has specific widget needs)
            let page = adw::PreferencesPage::new();

            let bing_group = adw::PreferencesGroup::new();
            bing_group.set_title("Bing");

            let market_row = adw::ComboRow::new();
            market_row.set_title("Market");
            market_row.set_model(Some(&market_list));
            market_row.set_selected(initial_index);

            let spotlight_group = adw::PreferencesGroup::new();
            spotlight_group.set_title("Spotlight");

            bing_group.add(&market_row);
            spotlight_group.add(&locale_entry);
            page.add(&bing_group);
            page.add(&spotlight_group);
            root.add(&page);
            root.set_title("Damask Preferences");
            root.set_modal(true);
            root.set_transient_for(None); // Set in show()

            // Connect apply/changes signals
            let sender_apply = sender.output_sender().clone();
            let markets_vec = markets.to_vec();
            market_row.connect_selected_notify(move |combo| {
                let sender = sender_apply.clone();
                let locale = locale_entry.text().to_string();
                let markets = markets_vec.clone();
                let idx = combo.selected() as usize;
                let market = markets.get(idx).map(|s| s.to_string()).unwrap_or_else(|| "en-US".to_string());
                sender.send(SettingsOutput::SettingsChanged { bing_market: market, spotlight_locale: locale }).unwrap();
            });

            let sender_locale = sender.output_sender().clone();
            let markets_vec2 = markets.to_vec();
            locale_entry.connect_changed(move |entry| {
                let sender = sender_locale.clone();
                let locale = entry.text().to_string();
                let markets = markets_vec2.clone();
                let market_idx = market_row.selected() as usize;
                let market = markets.get(market_idx).map(|s| s.to_string()).unwrap_or_else(|| "en-US".to_string());
                sender.send(SettingsOutput::SettingsChanged { bing_market: market, spotlight_locale: locale }).unwrap();
            });

            ComponentParts { model, widgets: () }
        }

        fn update(&mut self, message: Self::Input, _sender: ComponentSender<Self>, root: &Self::Root) {
            match message {
                SettingsInput::Show => {
                    root.present();
                }
            }
        }
    }
    ```

    NOTE: If using `SimpleComponent` + manual widget construction causes issues with `ComponentParts` expecting named widgets, switch to using the `view!` macro with `#[name]` attributes and reference widgets by name. The key point is the widget tree must match the UI-SPEC layout.

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| User input -> Bing market string | User-selected market code passed to Bing API as query parameter |
| User input -> Spotlight locale string | User-entered locale code passed to Spotlight API as query parameter |

## STRIDE Threat Register

| Threat ID | Category | Component | Severity | Disposition | Mitigation |
|-----------|----------|-----------|----------|-------------|------------|
| T-03-07 | I (Info Disclosure) | Locale/market in API URL | info | accept | Market codes (e.g. "en-US") and locale codes (e.g. "80217") are non-sensitive parameters sent over HTTPS to Microsoft APIs. No PII exposure. |
| T-03-08 | D (Denial of Service) | Invalid locale/market | info | accept | Invalid values cause API errors (SourceError) which are displayed as toasts. No system impact. The domain crate handles HTTP errors gracefully. |
| T-03-09 | T (Tampering) | Entry row input | info | accept | Entry row accepts arbitrary text but it is only used as an API query parameter. No code execution, no file system access. |
</threat_model>
